create table modify (
    id int primary key IDENTITY(1,1) NOT NULL, 
    type_target VARCHAR (10),
    target VARCHAR (30),
    type_modify VARCHAR(50),
    value VARCHAR(50),
);

create table class(
    id int PRIMARY KEY IDENTITY(1,1) NOT NULL,
    description VARCHAR(255),
    description_master VARCHAR(255),
);

create table class_modify(
    id int PRIMARY KEY IDENTITY(1,1) NOT NULL,
    class_id int FOREIGN KEY REFERENCES class(id),
    modify_id int FOREIGN KEY REFERENCES modify(id),
);

create table points(
  id int PRIMARY KEY IDENTITY(1,1) NOT NULL,
  survive int,
  special_goals int,
  quest int,
  player_kill int,
  total int, 
);

create table status (
    id int PRIMARY KEY IDENTITY(1,1) NOT NULL,
    mana_max int,
    mana_now int,
    stamina_max int,
    stamina_now int,
    life_max int,
    life_now int,
    xp_max int,
    xp_now int,
    non_used_points int,
);

create table atribute (
    id int PRIMARY KEY IDENTITY(1,1) NOT NULL,
    str int,
    dex int,
    con int,
    int int,
    cha int,
    luc int,
);

create table skill (
    id int PRIMARY KEY IDENTITY(1,1) NOT null,
    description VARCHAR(255),
    type_target VARCHAR(10),
    target VARCHAR(30),
    cost VARCHAR(50),
);

create table skill_level (
    id int PRIMARY KEY IDENTITY(1,1) NOT null,
    description VARCHAR(255),
    level int,
    skill_id int FOREIGN KEY REFERENCES skill(id)
);

create table player (
    id int PRIMARY KEY IDENTITY(1,1) NOT null,
    discord_id VARCHAR(255),
    name VARCHAR(100),
    level int,
    class_id int FOREIGN KEY REFERENCES class(id) DEFAULT NULL,
    stars int,
    atribute_id int FOREIGN KEY REFERENCES atribute(id) DEFAULT NULL,
    status_id int FOREIGN KEY REFERENCES status(id) DEFAULT NULL,
    points_id int FOREIGN KEY REFERENCES points(id) DEFAULT NULL,
    dead bit not null default 0
);

create table player_skill (
    id int PRIMARY KEY IDENTITY(1,1) NOT null,
    player_id int FOREIGN KEY REFERENCES player(id),
    skill_id int FOREIGN KEY REFERENCES skill(id),
    level int
);
